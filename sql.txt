select A.file_size, A.duration, substr(A.path, 5) from tbl_playitem A join tbl_playlist b on b.id = A.playlist_id
where A.path like '\HK\%'
order by A.file_size/A.duration

select A.file_size, A.duration, c.name, A.path || " " || b.name from tbl_albumitem A
join tbl_album c on c.id = A.album_id
join tbl_singer b on b.id = c.singer_id
left join tbl_playitem e on e.path = b.path || a.path
where b.path like '\HK\≈Æ∏Ë ÷%' and e.id isnull
order by A.file_size/A.duration


select * from tbl_singer A left join tbl_playitem B ON B.path like '%' || A.name || '%'
and B.path not like A.path || '\%'

select distinct A.path from tbl_playitem A join tbl_playitem B ON A.path = B.path and A.id <> B.id

select distinct A.path from tbl_albumitem A join tbl_albumitem B ON A.path = B.path and A.id <> B.id
